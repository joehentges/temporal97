[d3-force](https://d3js.org/d3-force) is a physics simulation module from D3.js. It does not store or render graphs — it takes an array of nodes and links and mutates them in place with `x` and `y` position properties over time. Use it when you want full control over rendering (canvas, SVG, WebGL) and just need a smart layout engine.

## Installation

```bash
npm install d3-force
npm install --save-dev @types/d3-force
```

## Building a simulation from live state

Extract the current live nodes and edges from temporal97 and feed them directly to d3-force:

```ts
import {
  forceSimulation,
  forceManyBody,
  forceLink,
  forceCenter,
  type SimulationNodeDatum,
  type SimulationLinkDatum,
} from 'd3-force';
import { TemporalGraph } from 'temporal97';

type NodeAttrs = { label: string };
type EdgeAttrs = { source: string; target: string };

const temporal = new TemporalGraph<NodeAttrs, EdgeAttrs>();

type SimNode = SimulationNodeDatum & { id: string; label: string };
type SimLink = SimulationLinkDatum<SimNode> & { id: string };

function buildSimulation(width: number, height: number) {
  const simNodes: SimNode[] = [...temporal.liveNodes()].map(([id, attrs]) => ({
    id,
    ...attrs,
  }));

  const nodeById = new Map(simNodes.map(n => [n.id, n]));

  const simLinks: SimLink[] = [...temporal.liveEdges()].map(([id, attrs]) => ({
    id,
    source: nodeById.get(attrs.source)!,
    target: nodeById.get(attrs.target)!,
  }));

  return forceSimulation(simNodes)
    .force('charge', forceManyBody().strength(-200))
    .force('link', forceLink<SimNode, SimLink>(simLinks).distance(60))
    .force('center', forceCenter(width / 2, height / 2));
}
```

## Rendering on canvas

Wire the simulation's `tick` event to a canvas render loop:

```ts
const canvas = document.getElementById('canvas') as HTMLCanvasElement;
const ctx = canvas.getContext('2d')!;
const { width, height } = canvas;

const sim = buildSimulation(width, height);

sim.on('tick', () => {
  ctx.clearRect(0, 0, width, height);

  // Draw edges
  for (const link of sim.force<any>('link').links()) {
    const s = link.source as SimNode;
    const t = link.target as SimNode;
    ctx.beginPath();
    ctx.moveTo(s.x!, s.y!);
    ctx.lineTo(t.x!, t.y!);
    ctx.strokeStyle = '#94a3b8';
    ctx.lineWidth = 1.5;
    ctx.stroke();
  }

  // Draw nodes
  for (const node of sim.nodes()) {
    ctx.beginPath();
    ctx.arc(node.x!, node.y!, 10, 0, Math.PI * 2);
    ctx.fillStyle = '#6366f1';
    ctx.fill();
    ctx.fillStyle = '#fff';
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(node.label, node.x!, node.y!);
  }
});
```

## Rebuilding after time travel

d3-force simulations are stateful — when you move the temporal cursor, rebuild the simulation from the new live state. Preserve existing node positions to avoid jarring resets:

```ts
let sim = buildSimulation(width, height);

function onTemporalSeek(snapshot: number) {
  const delta = temporal.seekTo(snapshot);

  // Cheap check — only rebuild if the graph structure changed
  if (
    delta.nodes.added.size > 0 ||
    delta.nodes.removed.size > 0 ||
    delta.edges.added.size > 0 ||
    delta.edges.removed.size > 0
  ) {
    // Capture current positions so nodes don't teleport
    const positions = new Map(sim.nodes().map(n => [n.id, { x: n.x, y: n.y }]));

    sim.stop();
    sim = buildSimulation(width, height);

    // Restore positions for nodes that still exist
    for (const node of sim.nodes()) {
      const prev = positions.get(node.id);
      if (prev) { node.x = prev.x; node.y = prev.y; }
    }

    sim.alpha(0.3).restart();
  }
}
```

## Combining with React

In a React component, rebuild the simulation inside a `useEffect` keyed on the current snapshot:

```tsx
import { useEffect, useRef } from 'react';

function GraphCanvas({ snapshot }: { snapshot: number }) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  useEffect(() => {
    temporal.seekTo(snapshot);
    const sim = buildSimulation(600, 400);
    const canvas = canvasRef.current!;
    const ctx = canvas.getContext('2d')!;

    sim.on('tick', () => renderFrame(ctx, sim));

    return () => { sim.stop(); };
  }, [snapshot]);

  return <canvas ref={canvasRef} width={600} height={400} />;
}
```

Change the `snapshot` prop and the component tears down the old simulation and starts a fresh one, preserving the temporal97 history intact.
